import { useState, useEffect } from "react";
import { cache } from "@/utils/cache";

export default function useDataWithCache(url) {
  const [data, setData] = useState(cache[url] || null);
  const [loading, setLoading] = useState(!cache[url]);

  useEffect(() => {
    if (cache[url]) return;

    let ignore = false;
    fetch(url)
      .then((res) => res.json())
      .then((json) => {
        if (!ignore) {
          cache[url] = json;
          setData(json);
          setLoading(false);
        }
      });

    return () => {
      ignore = true;
    };
  }, [url]);

  return { data, loading };
}
