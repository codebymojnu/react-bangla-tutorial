```javascript
const objectsArray = [
  { name: "Object 1", date: "2024-05-01" },
  { name: "Object 2", date: "2024-04-20" },
  { name: "Object 3", date: "2024-05-10" },
];

objectsArray.sort((a, b) => {
  const dateA = new Date(a.date);
  const dateB = new Date(b.date);
  return dateA - dateB;
});

console.log(objectsArray);

// [
//   { name: 'Object 2', date: '2024-04-20' },
//   { name: 'Object 1', date: '2024-05-01' },

// ]
```
